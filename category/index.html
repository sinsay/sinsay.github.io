简介

因为 C 语言本身只提供了 Array 数组，在需要动态添加删除元素时，需要使用者自行扩充长度，造成使用上的繁琐，鉴于这点许多高级语言都在语言层间上添加了更加高级的 List 类型，让使用者避开内存以及列表长度的管理。

当然，List 跟 Array 本身还是有区别的，因为 Array 的访问效率是 O(1)，而 List 一般的实现都是链表，所以除了首端跟末端，其他的访问效率都是 O(N)，这些具体的区别这里就不一一说明了，但会在接口上进行详解，如果需要更接近 Array 访问效率的列表，可以考虑使用 Vector，他的访问效率跟 Array 是一样的，只是在 Array 上增加了自动扩充长度的功能。

实现

定义

下面是 List 的结构定义

// List 的节点
typedef struct listNode {
    struct listNode *prev;
    struct listNode *nex;
    void *value;
} listNode;

// 对于节点的封装，用以迭代节点
typedef struct listIter {
    listNode *next;
    int direction;
} listIter;

// List 结构的定义
typedef struct list {
    listNode *head;
    listNode *tail;
    void *(*dup)(void *ptr);
    void *(free)(void *ptr);
    int (*match)(void *ptr, void *key);
    unsigned long len;
} list;

一些便捷的宏定义

对于一些简单的操作及实现，用宏来进行包装，已避免函数调用等开销

#define listLength(l) ((l)->len)
#define listFirst(l) ((l)->head)
#define listLast(l) ((l)->tail)
#define listPrevNode(n) ((n)->prev)
#define listNextNode(n) ((n)->next)
#define listNodeValue(n) ((n)->value)

#define listSetDupMethod(l, m) ((l)->dup = (m))
#define listSetFreeMethod(l, m) ((l)->free = (m))
#define listSetMatchMethod(l, m) ((l)->match = (m))

#define listGetDupMethod(l) ((l)->dup)
#define listGetFreeMethod(l) ((l)->free)
#define listGetMatchMethod(l) ((l)->match);

列表的创建

创建一个列表较简单，初始化 list 结构体而已

list *listCreate(void) {
    struct list *list;
    
    if ((list = zmalloc(sizeof(*list))) == NULL)
        return NULL;
    list->head = list->tail = NULL;
    list->len = 0;
    list->dup = NULL;
    list->free = NULL;
    list->match = NULL;
    return list;
}

添加一个元素

List 对操作节点提供了多种方式的操作，比如在首部或尾部插入新节点，指定位置插入节点，指定节点后插入等，这些操作的时间复杂度都是不同的，这里我们就各个操作都进行一下说明。

首部跟尾部插入 复杂度 O(1)

因为 List 的实现是双向链表，并且同时保存了 head 跟 tail 指针，所以在这两个位置插入新的节点时，可以在 O(1) 的时间内完成，下面是代码

list *listAddNodeHead(list *list, void *value) {
    listNode *node;
    
    if ((node = zmalloc(sizeof(*node))) == NULL)
        return NULL;
    node->value = value;
    // 如果 List 是空的，则 head 跟 tail 都指向新的节点
    // 并且，它的前后节点都为空
    if (list->len == 0) { 
        list->head = list->tail = node;
        node->prev = node->next = NULL;
    } else {
        // 否则让新节点成为头结点，并让原有的头节点成为新节点的下一节点
        node->prev = NULL;
        node->next = list->head;
        list->head->prev = node;
        list->head = node;
    }
    list->len++;
    return list;
}

list listAddNodeTail(list *list, void *value) {
    listNode *node;
    
    if ((node = zmalloc(sizeof(*node))) == NULL)
        return NULL;
        
    node->value = value;
    if (list->len == 0) {
        list->head = list->tail = node;
        node->prev = node->next = NULL;
    } else {
        node->prev = list-tail;
        node->next = NULL;
        list->tail->next = node;
        list->tail = node;
    }
    list->len++;
    return list;
}

查找指定节点后从节点位置插入

指定节点插入的时间复杂度是 O(1), 当然，要插入到指定节点后，那首先肯定要找到这个节点，找到节点的时间复杂度取决于插入的位置，而如果插入的位置超过了 List 本身的长度，则视为出错。

首先是查找节点的过程，有两种方式

  • 第一种，指定索引，从指定位置开始迭代，直到遇见所需的节点。查找指定的位置支持正序以及反序，正数表示从列表头开始查找，而负数则表示从列表结尾开始查找，如 -1 表示 List 的最后一个节点，-2 表示最后一个节点的上一个节点。
  • 第二种，指定 key，也就是查找跟某个值相等的节点

这两种方式都需要有一种遍历列表的机制，这个具体的机制，可以跳过来先看 如何遍历整个列表。

然后是从指定的节点插入新节点。

listNode *listIndex(list *list, int index) {
    listNode *n;
    
    if (index < 0) {
        index = (-index)-1;
        n = list->tail;
        while (index-- && n) n = n->prev;
    } else {
        n = list->head;
        while (index-- && n) n = n->next;
    }
    
    return n;
}

listNode *listSearchKey(list *list, void *key) {
    listIter *iter;
    listNode *node;
    
    // 从头开始查找
    iter = listGetIterator(list, AL_START_HEAD);
    while ((node = listNext(iter)) != NULL) {
        if (list->match) {
            if (list->match(node->value, key)) {
                listReleaseIterator(iter);
                return node;
            }
        } else {
            if (key == node->value) {
                listReleaseIterator(iter);
                return node;
            }
        }
    }
    listReleaseIterator(iter);
    return NULL;
}

list *listInsertNode(list *list, listNode *old_node, void *value, int after) {
    listNode *node;
    
    if ((node = zmalloc(sizeof(*node))) == NULL)
        return NULL;
        
    node->value = value;
    // 插入到指定节点的后面
    if (after) {
        node->prev = oldnode;
        node->next = oldnode->next;
        if (list->tail == old_node) { 
            // 如果插入的位置是最后一个节点，则更新新节点为最后一个节点
            list->tail = node;
        }
    } else { // 插入到指定节点的前面
        node->next = oldnode;
        node->prev = oldnode->prev;
        if (list->head == old_node) {
            list->head = node;
        }
    }
    
    // 然后调整关联的节点
    if (node->prev != NULL) {
        node->prev->next = node;
    }    
    if (node->next != NULL) {
        node->next->prev = node;
    }
    list->len++;
    return list;
}

删除指定节点

void listDelNode(list *list, listNode *node) {
    if (node->prev)
        node->prev->next = node->next;
    else
        list->head = node->next;
    if (node->next)
        node->next->prev = node->prev;
    else
        list->tail = node->prev;
        
    if (list->free) list->free(node->value);
    zfree(node);
    list->len--;
}

遍历所有节点

List 封装了一个迭代器类型用于遍历所有的节点，使用迭代器的好处是区分开了 List 具体的实现，这样即使是以后 List 有任何的修改，都不会影响到使用者。

首先看如何得到一个迭代器。

listIter *listGetIterator(list *list, int direction) {
    listIter *iter;
    if ((iter = zmalloc(sizeof(*iter))) == NULL)
        return NULL;
    if (direction == AL_START_HEAD)
        iter->next = list->head;
    else
        iter->next = list->tail;
    iter->direction = direction;
    return iter;
}

我们看到他构建了一个 listIter 后根据迭代的方向把 head 或 tail 给予了 iter 的 next，这就保证了之后的遍历可以使用 listNext 来继续遍历。
而根据迭代的方向，listNext 则不断的返回当前迭代器指向的下一个节点，并将自身指向下一个节点。

listNode *listNext(listIter *iter) {
    listNode *current = iter->next;
    if (current != NULL) {
        if (iter->direction == AL_START_HEAD)
            iter->next = current->next;
        else
            iter->next = current->prev;
    }
    return current;
}

释放一个迭代器

void listReleaseIterator(listIter *iter) {
    zfree(iter);
}

重置当前的迭代器

重置迭代器，根据遍历的方向使其指向开头或者结尾。

void listRewind(list *list, listIter *li) {
    li->next = list->head;
    li->direction = AL_START_HEAD;
}

void listRewindToTail(list *list, listIter *li) {
    li->next = list->tail;
    li->direction = AL_START_TAIL;
}

释放当前 List

void listRelease(list *list) {
    unsigned long len;
    listNode *current, *next;
    
    current = list->head;
    len = list->len;
    while (len--) {
        next = current->next;
        if (list->free) list->free(current->value);
        zfree(current);
        current = next;
    }
    zfree(list);
}

其他的 List 操作

复制列表

list *listDup(list *orig) {
    list *copy;
    listIter *iter;
    listNode *node;
    
    if ((copy = listCreate()) == NULL)
        return NULL;
        
    // 这里为什么不使用预先定义好的宏？？？
    copy->dup = orig->dup;
    copy->free = orig->free;
    copy->match = orig->match;
    iter = listGetIterator(orig, AL_START_HEAD);
    
    while ((node = listNext(iter)) != NULL) {
        void *value;
        
        if (copy->dup) {
            value = copy->dup(node->value);
            if (value == NULL) {
                listRelease(copy);
                listReleaseIterator(iter);
                return NULL;
            }
        } else
            value = node->value;
            
        if (listAddNodeTail(copy, value) == NULL) {
            listRelease(copy);
            listReleaseIterator(iter);
            return NULL;
        }
    }
    
    listReleaseIterator(iter);
    return copy;
}

旋转列表

将 List 进行旋转，以原本的 tail 作为 head, 相当于把整个 List 进行了一下左旋。

void listRotate(list *list) {
    listNode *tail = list->tail;
    
    if (listLength(list) <= 1) return;
    
    // Detach current tail
    list->tail = tail->prev;
    list->tail->next = NULL;
    // move it as head
    list->head->prev = tail;
    tail->prev = NULL;
    tail->next = list->head;
    list-head = tail;
}
